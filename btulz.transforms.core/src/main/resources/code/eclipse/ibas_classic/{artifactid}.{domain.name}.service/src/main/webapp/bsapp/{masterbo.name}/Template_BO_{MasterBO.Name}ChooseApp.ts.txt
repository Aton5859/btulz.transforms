import * as ibas from "ibas/index";
import * as bo from "../../borep/bo/index";
import { BORepository${Domain.getName()} } from "../../borep/BORepositories";
import { ${BO.getName()}EditApp } from "./${BO.getName()}EditApp";

/** 选择应用-${BO.getDescription()} */
export class ${BO.getName()}ChooseApp extends ibas.BOChooseService<I${BO.getName()}ChooseView, bo.${BO.getName()}> {

    /** 应用标识 */
    static APPLICATION_ID: string = "${ID.newUUID()}";
    /** 应用名称 */
    static APPLICATION_NAME: string = "${Domain.Name().toLowerCase()}_app_${BO.getName().toLowerCase()}_choose";
    /** 业务对象编码 */
    static BUSINESS_OBJECT_CODE: string = bo.${BO.getName()}.BUSINESS_OBJECT_CODE;
    /** 构造函数 */
    constructor() {
        super();
        this.id = ${BO.getName()}ChooseApp.APPLICATION_ID;
        this.name = ${BO.getName()}ChooseApp.APPLICATION_NAME;
        this.boCode = ${BO.getName()}ChooseApp.BUSINESS_OBJECT_CODE;
        this.description = ibas.i18n.prop(this.name);
    }
    /** 注册视图 */
    protected registerView(): void {
        super.registerView();
        // 其他事件
    }
    /** 启动应用 */
    run(...args: any[]): void {
        let criteria: ibas.ICriteria = arguments[0];
        if (ibas.objects.instanceOf(criteria, ibas.Criteria)) {
            // 传入了查询，则使用
            this.view.query(criteria);
        } else {
            super.run();
        }
    }
    /** 视图显示后 */
    protected viewShowed(): void {
        // 视图加载完成
    }
    /** 查询数据 */
    protected fetchData(criteria: ibas.ICriteria): void {
        this.busy(true);
        let that: this = this;
        let boRepository: BORepository${Domain.getName()} = new BORepository${Domain.getName()}();
        boRepository.fetch${BO.getName()}({
            criteria: criteria,
            onCompleted(opRslt: ibas.IOperationResult<bo.${BO.getName()}>): void {
                try {
                    if (opRslt.resultCode !== 0) {
                        throw new Error(opRslt.message);
                    }
                    if (opRslt.resultObjects.length === 1
                        && ibas.config.get(ibas.CONFIG_ITEM_AUTO_CHOOSE_DATA, true)) {
                        // 仅一条数据，直接选择
                        that.chooseData(opRslt.resultObjects);
                    } else {
                        if (!that.isViewShowed()) {
                            // 没显示视图，先显示
                            that.show();
                        }
                        that.view.showData(opRslt.resultObjects);
                        that.busy(false);
                    }
                } catch (error) {
                    that.messages(error);
                }
            }
        });
        this.proceeding(ibas.emMessageType.INFORMATION, ibas.i18n.prop("shell_fetching_data"));
    }
    /** 新建数据 */
    protected newData(): void {
        // 关闭自身
        this.destroy();
        // 调用编辑应用
        let app: ${BO.getName()}EditApp = new ${BO.getName()}EditApp();
        app.navigation = this.navigation;
        app.viewShower = this.viewShower;
        app.run();
    }
}
/** 视图-${BO.getDescription()} */
export interface I${BO.getName()}ChooseView extends ibas.IBOChooseView {
    /** 显示数据 */
    showData(datas: bo.${BO.getName()}[]): void;
}
/** ${BO.getDescription()}选择服务映射 */
export class ${BO.getName()}ChooseServiceMapping extends ibas.BOChooseServiceMapping {
    /** 构造函数 */
    constructor() {
        super();
        this.id = ${BO.getName()}ChooseApp.APPLICATION_ID;
        this.name = ${BO.getName()}ChooseApp.APPLICATION_NAME;
        this.boCode = ${BO.getName()}ChooseApp.BUSINESS_OBJECT_CODE;
        this.description = ibas.i18n.prop(this.name);
    }
    /** 创建服务并运行 */
    create(): ibas.IService<ibas.IServiceContract> {
        return new ${BO.getName()}ChooseApp();
    }
}
